Categories & collections are quite similar to search, except that there's no search field/search modal. Instead of being based on a search query, they're based on a so-called `query_id` which is an id representing what category or collection to show.

<Frame caption="A default `CategoryPage` (with customised styling)">
    <img src="/images/category_page_overview.svg" ref={async element => {
        // Fix background color in dark mode while still having cool, transparent light mode background
        if(!element) return;
        const load_utilishared = () => {
            const script = document.createElement("script");
            script.type = "module";
            script.append(`
            import * as utilishared from "https://esm.run/@depict-ai/utilishared@latest/dist/module.js";
            window.utilishared = utilishared;
            window?.resolve_utilishared();
            `)
            document.head.append(script);
        }

        if(!window.utilishared) {
            await new Promise(resolve => {
                window.resolve_utilishared = resolve;
                load_utilishared();
            })
        }

        const { style } = element;
        const update_styling = () => {
            const dark = document.documentElement.classList.contains("dark");
            style.background = dark ? "white" : "";
            style.padding = dark ? "5px" : "";
            style.borderRadius = dark ? "0.75rem" : "";
        };
        const m_o = new MutationObserver(utilishared.catchify(update_styling));
        m_o.observe(document.documentElement, {
            attributes: true,
            attributeFilter: ["class"]
        })
        update_styling();
        utilishared.observer.onremoved(element, () => m_o.disconnect());
    }} />
</Frame>
